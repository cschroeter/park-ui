## Usage

```tsx
import { TagsInput } from '@/components/ui'
```

```tsx
<TagsInput.Root>
  <TagsInput.Label>Tags Input</TagsInput.Label>
  <TagsInput.Control>
    <TagsInput.Context>
      {({ value }) =>
        value.map((item, index) => (
          <TagsInput.Item key={index} index={index} value={item}>
            <TagsInput.ItemPreview>
              <TagsInput.ItemText>{item}</TagsInput.ItemText>
              <TagsInput.ItemDeleteTrigger />
            </TagsInput.ItemPreview>
            <TagsInput.ItemInput />
          </TagsInput.Item>
        ))
      }
    </TagsInput.Context>
    <TagsInput.Input placeholder="Add tag..." />
  </TagsInput.Control>
</TagsInput.Root>
```
