## Usage

```tsx
import { Select } from '@/components/ui'
```

```tsx
<Select.Root>
  <Select.Label>Select Label</Select.Label>
  <Select.Control>
    <Select.Trigger>
      <Select.ValueText placeholder="Select option..." />
      <Select.Indicator />
    </Select.Trigger>
  </Select.Control>
  <Select.Positioner>
    <Select.Content>
      <Select.Item item="option1">
        <Select.ItemText>Option 1</Select.ItemText>
        <Select.ItemIndicator />
      </Select.Item>
    </Select.Content>
  </Select.Positioner>
</Select.Root>
```
