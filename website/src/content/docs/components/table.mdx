## Usage

```tsx
import { Table } from '@/components/ui'
```

```tsx
<Table.Root>
  <Table.Caption />
  <Table.Head>
    <Table.Row>
      <Table.Header />
    </Table.Row>
  </Table.Head>
  <Table.Body>
  <Table.Row>
    <Table.Cell />
    </Table.Row>
  </Table.Body>
  <Table.Foot>
    <Table.Row>
      <Table.Cell />
    </Table.Row>
  </Table.Foot>
</Table.Root>
```
