## Usage

```tsx
import { InputGroup } from '@/components/ui'
```

```tsx
<InputGroup.Root>
  <InputGroup.Addon />
  <InputGroup.Control startElement={} endElement={}>
    <Input />
  </InputGroup.Control>
  <InputGroup.Addon />
</InputGroup.Root>
```
