## Usage

```tsx
import { Splitter } from '@/components/ui'
```

```tsx
<Splitter.Root size={[
  { id: 'a', size: 50 },
  { id: 'b', size: 50 },
]}>
  <Splitter.Panel id="a">Panel A</Splitter.Panel>
  <Splitter.ResizeTrigger id="a:b" />
  <Splitter.Panel id="b">Panel B</Splitter.Panel>
</Splitter.Root>
```
