## Usage

```tsx
import { Skeleton, SkeletonCircle, SkeletonText } from "@/components/ui"
```

```tsx
<Stack gap="6">
  <HStack>
    <SkeletonCircle boxSize="10" />
    <SkeletonText noOfLines={2} />
  </HStack>
  <Skeleton height="48" />
</Stack>
```
