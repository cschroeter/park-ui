## Usage

```tsx
import { Combobox } from '@/components/ui'
```

```tsx
<Combobox.Root>
  <Combobox.Control>
    <Combobox.Input placeholder="Select option..." />
    <Combobox.Trigger>
      <Combobox.ClearTrigger />
    </Combobox.Trigger>
  </Combobox.Control>
  <Combobox.Positioner>
    <Combobox.Content>
      <Combobox.Item item="option1">Option 1</Combobox.Item>
    </Combobox.Content>
  </Combobox.Positioner>
</Combobox.Root>
```
