## Usage

```tsx
import { Tooltip } from '@/components/ui'
```

```tsx
<Tooltip.Root>
  <Tooltip.Trigger>Hover me</Tooltip.Trigger>
  <Tooltip.Positioner>
    <Tooltip.Content>
      <Tooltip.Arrow>
        <Tooltip.ArrowTip />
      </Tooltip.Arrow>
      Tooltip content
    </Tooltip.Content>
  </Tooltip.Positioner>
</Tooltip.Root>
```
