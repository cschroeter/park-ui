## Usage

```tsx
import { PinInput } from '@/components/ui'
```

```tsx
<PinInput.Root>
  <PinInput.Label>Pin Input</PinInput.Label>
  <PinInput.Control>
    {Array.from({ length: 4 }, (_, index) => (
      <PinInput.Input key={index} index={index} />
    ))}
  </PinInput.Control>
</PinInput.Root>
```
