## Usage

```tsx
import { RatingGroup } from '@/components/ui'
```

```tsx
<RatingGroup.Root count={5} value={3}>
  <RatingGroup.Label>Rating</RatingGroup.Label>
  <RatingGroup.Control>
    <RatingGroup.Context>
      {({ items }) =>
        items.map((item) => (
          <RatingGroup.Item key={item} index={item}>
            <RatingGroup.ItemIndicator />
          </RatingGroup.Item>
        ))
      }
    </RatingGroup.Context>
  </RatingGroup.Control>
</RatingGroup.Root>
```
